releases/FIRMWARE_mayhem_v2.4.0.zip              # USB flash bundle (8 MB)

## Source repositories (repos/)
repos/hackrf              # Official HackRF host tools + libhackrf
repos/mayhem-firmware     # PortaPack Mayhem (build + docs)
repos/portapack-hackrf    # Original PortaPack reference
repos/urh                 # Universal Radio Hacker source
repos/hacktv              # Analog TV transmitter source

## Built from source (build/)
build/hackrf-tools/src/hackrf_sweep   # Spectrum sweep (also in hackrf repo)
build/hackrf-tools/src/hackrf_*       # All host utilities
build/libhackrf/src/libhackrf.so      # Newer lib than apt (0.10.0)

## Helper scripts (scripts/)
source scripts/env.sh          # PATH + aliases
scripts/setup-udev.sh          # USB permissions (plugdev)
scripts/download-mayhem.sh     # Re-fetch release assets
scripts/prepare-sdcard.sh      # Extract SD card files
scripts/flash-mayhem.sh        # Extract firmware flash bundle
scripts/build-mayhem.sh        # Compile Mayhem from repos/mayhem-firmware
scripts/launch-urh.sh          # Start URH GUI
